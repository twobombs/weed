71  opt.t += 1;

74  (real1)(ONE_R1 - std::pow((real1_s)opt.beta1, (real1_s)opt.t));
75  const real1 bias_correction2 =
76  (real1)(ONE_R1 - std::pow((real1_s)opt.beta2, (real1_s)opt.t));
77 
78  for (auto &p : params) {
79  AdamState &s = opt.state[p];
80  TensorPtr g = p->grad;
81 
82  // m = beta1 * m + (1 - beta1) * g
83  s.m = opt.beta1 * s.m + (ONE_R1 - opt.beta1) * g;
84 
85  // v = beta2 * v + (1 - beta2) * g * g
86  s.v = opt.beta2 * s.v + (ONE_R1 - opt.beta2) * g * g;
87  // i.e., v += (1-beta2) * (g âŠ™ g)
88

92  // p -= lr * tmp / (sqrt(tmp2) + eps)
