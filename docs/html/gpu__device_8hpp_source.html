160  bool LockSync(BufferPtr buffer, const size_t &sz, void *array,
161  bool allow_lock = true) {
162  EventVecPtr waitVec = ResetWaitEvents();
163 
164  if (allow_lock && device_context->use_host_mem) {
165  // Needs unlock
166  tryOcl("Failed to map buffer", [&] {
167  cl_int error;
168  queue.enqueueMapBuffer(*buffer, CL_TRUE, CL_MAP_READ, 0U, sz,
169  waitVec.get(), nullptr, &error);
170  return error;
171  });
172  wait_refs.clear();
173  } else {
174  // Just a copy: don't need to unlock
175  tryOcl("Failed to read buffer", [&] {
176  return queue.enqueueReadBuffer(*buffer, CL_TRUE, 0U, sz, array,
177  waitVec.get());
178  });
179  wait_refs.clear();
180  }
181 
182  return device_context->use_host_mem;
183  }
