204  static std::vector<tcapint>
205  full_contiguous_stride(const std::vector<tcapint> &shp) {
206  if ((shp.size() == 1U) && (shp[0U] == 1U)) {
207  return std::vector<tcapint>{0U};
208  }
209 
210  std::vector<tcapint> g_stride(shp.size());
211  tcapint max_index = 1U;
212  for (size_t i = 0U; i < shp.size(); ++i) {
213  g_stride[i] = max_index;
214  max_index *= shp[i];
215  }
216 
217  return g_stride;
218  }
